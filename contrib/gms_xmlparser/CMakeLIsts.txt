#This is the main CMAKE for build all gms_xmlparser.
# gms_xmlparser
AUX_SOURCE_DIRECTORY(${CMAKE_CURRENT_SOURCE_DIR} TGT_gms_xmlparser_SRC)
set(TGT_gms_xmlparser_INC 
    ${PROJECT_OPENGS_DIR}/contrib/gms_xmlparser
    ${PROJECT_OPENGS_DIR}/contrib
)

set(gms_xmlparser_DEF_OPTIONS ${MACRO_OPTIONS})
set(gms_xmlparser_COMPILE_OPTIONS ${OPTIMIZE_OPTIONS} ${OS_OPTIONS} ${PROTECT_OPTIONS} ${WARNING_OPTIONS} ${LIB_SECURE_OPTIONS} ${CHECK_OPTIONS})
set(gms_xmlparser_LINK_OPTIONS ${LIB_LINK_OPTIONS})
add_shared_libtarget(gms_xmlparser TGT_gms_xmlparser_SRC TGT_gms_xmlparser_INC "${gms_xmlparser_DEF_OPTIONS}" "${gms_xmlparser_COMPILE_OPTIONS}" "${gms_xmlparser_LINK_OPTIONS}")
set_target_properties(gms_xmlparser PROPERTIES PREFIX "")

install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/gms_xmlparser.control
    DESTINATION share/postgresql/extension/
) 
install(FILES ${CMAKE_CURRENT_SOURCE_DIR}/gms_xmlparser--1.0.sql
    DESTINATION share/postgresql/extension/
)
install(TARGETS gms_xmlparser DESTINATION lib/postgresql)
